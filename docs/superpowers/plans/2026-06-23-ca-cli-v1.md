# `ca` CLI — v1 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build `ca`, a developer CLI that discovers `@flow` and `Agent(...)` agents in a module and gives one selection grammar to inspect (`ls`/`show`/`graph`), run locally (`run`), gate (`lint`/`test`), and observe (`trace`/`doctor`) any slice — as porcelain over the existing `composable_agents` plumbing.

**Architecture:** A new strict-typed subpackage `composable_agents/ca/`. Discovery is **hybrid**: a cheap AST scan finds agents + intra-`@flow` cross-agent edges with no import; an isolated **subprocess resolver** imports user code only when a verb needs the runnable IR (`run`/`lint`). A `Module` model is the single source of truth every verb consumes through a shared selector engine. Trace trees render from in-memory projection events (offline); Langfuse deep-links are built from the existing stable `trace_id_for(run_id)`.

**Tech Stack:** Python ≥3.12, Typer (new `cli` extra; core stays PyYAML-only), the existing `composable_agents` APIs (`define.FlowDef`, `dag.Graph`, `ir.Node`, `deploy.deploy`, `execution.interpreter`, `projection`, `execution.langfuse`), `pytest` + `capsys` for tests.

```python
app = typer.Typer(add_completion=True, no_args_is_help=True, help="Developer CLI for composable-agents modules.")

@app.callback(invoke_without_command=True)
def _root(
    ctx: typer.Context,
    version: bool = typer.Option(False, "--version", help="Print ca version and exit."),
) -> None:
    if version:
        typer.echo(f"ca {VERSION}")
        raise typer.Exit(0)


def main(argv: list[str] | None = None) -> int:
    """Entry point. Returns a process exit code (does not call sys.exit)."""
    try:
        app(args=argv, standalone_mode=False)
    except SystemExit as exc:  # argparse-style callers expect an int
        return int(exc.code or 0)
    except typer.Exit as exc:
        return int(exc.exit_code)
    return 0
```

```python
# composable_agents/ca/config.py
from __future__ import annotations

import tomllib
from dataclasses import dataclass, field
from pathlib import Path
from typing import Any


@dataclass(frozen=True)
class JulepConfig:
    root: Path
    src: list[str] = field(default_factory=list)
    exclude: list[str] = field(default_factory=list)
    tags: dict[str, list[str]] = field(default_factory=dict)
    fail_severity: str = "error"


def _read_toml(path: Path) -> dict[str, Any]:
    if not path.is_file():
        return {}
    with path.open("rb") as fh:
        return tomllib.load(fh)


def load_config(root: str | Path) -> JulepConfig:
    """Read [tool.julep] from pyproject.toml, then overlay a sibling julep.toml if present."""
    root = Path(root)
    pyproject = _read_toml(root / "pyproject.toml").get("tool", {}).get("ca", {})
    julep_toml = _read_toml(root / "julep.toml")
    merged: dict[str, Any] = {**pyproject, **julep_toml}
    gates = {**pyproject.get("gates", {}), **julep_toml.get("gates", {})}
    return JulepConfig(
        root=root,
        src=list(merged.get("src", [str(root)])),
        exclude=list(merged.get("exclude", [])),
        tags={k: list(v) for k, v in merged.get("tags", {}).items()},
        fail_severity=str(gates.get("fail_severity", "error")),
    )
```

```python
# composable_agents/ca/discover.py
from __future__ import annotations

import ast
from dataclasses import dataclass, field
from pathlib import Path

from composable_agents.ca.config import JulepConfig


@dataclass(frozen=True)
class AgentInfo:
    name: str
    kind: str            # "flow" | "agent"
    file: str            # absolute path
    line: int
    calls: list[str] = field(default_factory=list)  # names of other agents this one calls


def _iter_files(cfg: JulepConfig) -> list[Path]:
    files: list[Path] = []
    for entry in cfg.src:
        base = (cfg.root / entry)
        if base.is_file() and base.suffix == ".py":
            files.append(base)
        elif base.is_dir():
            files.extend(p for p in base.rglob("*.py") if p.name != "__init__.py")
    excluded = set()
    for pat in cfg.exclude:
        excluded.update(cfg.root.glob(pat))
        for entry in cfg.src:
            excluded.update((cfg.root / entry).rglob(pat))
    return [f for f in files if f not in excluded]


def _is_flow_decorator(dec: ast.expr) -> bool:
    target = dec.func if isinstance(dec, ast.Call) else dec
    return (isinstance(target, ast.Name) and target.id == "flow") or (
        isinstance(target, ast.Attribute) and target.attr == "flow"
    )


def _agent_call_name(node: ast.expr) -> bool:
    func = node.func if isinstance(node, ast.Call) else None
    return bool(func and isinstance(func, ast.Name) and func.id == "Agent")


def scan_agents(cfg: JulepConfig) -> list[AgentInfo]:
    """Find top-level @flow functions and `x = Agent(...)` assignments via AST. No import."""
    out: list[AgentInfo] = []
    flow_names: set[str] = set()
    raw: list[tuple[ast.AST, str, str, str, int]] = []  # (node, name, kind, file, line)

    for path in _iter_files(cfg):
        try:
            tree = ast.parse(path.read_text(encoding="utf-8"))
        except SyntaxError:
            continue
        for node in tree.body:
            if isinstance(node, (ast.FunctionDef, ast.AsyncFunctionDef)) and any(
                _is_flow_decorator(d) for d in node.decorator_list
            ):
                flow_names.add(node.name)
                raw.append((node, node.name, "flow", str(path), node.lineno))
            elif isinstance(node, ast.Assign) and isinstance(node.value, ast.Call) and _agent_call_name(node.value):
                name = _agent_name_kwarg(node.value) or (
                    node.targets[0].id if isinstance(node.targets[0], ast.Name) else "agent"
                )
                raw.append((node, name, "agent", str(path), node.lineno))

    for node, name, kind, file, line in raw:
        calls = sorted({c for c in _called_names(node) if c in flow_names and c != name})
        out.append(AgentInfo(name=name, kind=kind, file=file, line=line, calls=calls))
    return out


def _agent_name_kwarg(call: ast.Call) -> str | None:
    for kw in call.keywords:
        if kw.arg == "name" and isinstance(kw.value, ast.Constant) and isinstance(kw.value.value, str):
            return kw.value.value
    return None


def _called_names(node: ast.AST) -> set[str]:
    names: set[str] = set()
    for sub in ast.walk(node):
        if isinstance(sub, ast.Call) and isinstance(sub.func, ast.Name):
            names.add(sub.func.id)
    return names
```

```python
# composable_agents/ca/model.py
from __future__ import annotations

from dataclasses import dataclass, field

from composable_agents.ca.config import JulepConfig
from composable_agents.ca.discover import scan_agents


@dataclass(frozen=True)
class Agent:
    name: str
    kind: str
    file: str
    line: int
    tags: list[str] = field(default_factory=list)
    calls: list[str] = field(default_factory=list)


@dataclass(frozen=True)
class Module:
    root: str
    agents: list[Agent]

    def by_name(self, name: str) -> Agent:
        for a in self.agents:
            if a.name == name:
                return a
        raise KeyError(name)


def build_module(cfg: JulepConfig) -> Module:
    agents = [
        Agent(
            name=info.name,
            kind=info.kind,
            file=info.file,
            line=info.line,
            tags=list(cfg.tags.get(info.name, [])),
            calls=list(info.calls),
        )
        for info in sorted(scan_agents(cfg), key=lambda i: i.name)
    ]
    return Module(root=str(cfg.root), agents=agents)
```

```python
# composable_agents/ca/_resolve_child.py
"""Run inside the user's interpreter (separate process). Imports user modules,
finds the named agent, prints {ir, param_names} or {error} as JSON to stdout."""
from __future__ import annotations

import importlib
import json
import sys
from pathlib import Path


def _discover_module_names(root: str, src: list[str]) -> list[str]:
    names: list[str] = []
    root_path = Path(root)
    for entry in src:
        base = root_path / entry
        for py in ([base] if base.is_file() else base.rglob("*.py")):
            if py.name == "__init__.py":
                continue
            rel = py.relative_to(root_path).with_suffix("")
            names.append(".".join(rel.parts))
    return names


def main() -> int:
    payload = json.loads(sys.argv[1])
    root, src, target = payload["root"], payload["src"], payload["name"]
    sys.path.insert(0, root)
    from composable_agents.define import FlowLike  # type: ignore[attr-defined]

    found = None
    for modname in _discover_module_names(root, src):
        try:
            mod = importlib.import_module(modname)
        except Exception:  # noqa: BLE001 — a bad module shouldn't kill resolution of others
            continue
        for attr in vars(mod).values():
            if isinstance(attr, FlowLike) and getattr(attr, "name", None) == target:
                found = attr
                break
        if found is not None:
            break

    if found is None:
        print(json.dumps({"error": f"agent {target!r} not found"}))
        return 0
    node = found.to_ir()
    print(json.dumps({"ir": node.to_json(), "name": target}))
    return 0


if __name__ == "__main__":
    raise SystemExit(main())
```

```python
# composable_agents/ca/resolve.py
from __future__ import annotations

import json
import subprocess
import sys
from dataclasses import dataclass
from typing import Any

from composable_agents.ca.config import JulepConfig


@dataclass(frozen=True)
class ResolvedAgent:
    name: str
    ir: dict[str, Any]
    error: str | None = None


def resolve_agent(cfg: JulepConfig, name: str, *, timeout: float = 30.0) -> ResolvedAgent:
    """Import the user's module in a subprocess and return the agent's serialized IR."""
    arg = json.dumps({"root": str(cfg.root), "src": cfg.src, "name": name})
    try:
        proc = subprocess.run(
            [sys.executable, "-m", "composable_agents.ca._resolve_child", arg],
            capture_output=True, text=True, timeout=timeout, cwd=str(cfg.root),
```

```python
# composable_agents/ca/select.py
from __future__ import annotations

import fnmatch
from pathlib import Path

from composable_agents.ca.gitstate import modified_agent_files
from composable_agents.ca.model import Agent, Module


def _match_method(module: Module, token: str, *, state_ref: str) -> set[str]:
    """Resolve one selector token to a set of agent names."""
    if token.startswith("tag:"):
        want = token[4:]
        return {a.name for a in module.agents if want in a.tags}
    if token.startswith("path:"):
        glob = token[5:]
        return {
            a.name for a in module.agents
            if fnmatch.fnmatch(str(Path(a.file).relative_to(module.root)), glob)
        }
    if token == "state:modified":
        changed = modified_agent_files(module.root, state_ref)
        return {a.name for a in module.agents if str(Path(a.file).resolve()) in changed}
    if token == "result:fail":
        from composable_agents.ca.runcache import failed_agents  # late import: optional cache
        failed = failed_agents(module.root)
        return {a.name for a in module.agents if a.name in failed}
    return {a.name for a in module.agents if a.name == token}  # bare name


def _eval_expr(module: Module, expr: str, *, state_ref: str) -> set[str]:
    """Space = union of comma-groups; within a comma-group, intersection of tokens."""
    if not expr.strip():
        return {a.name for a in module.agents}
    result: set[str] = set()
    for group in expr.split():
        tokens = [t for t in group.split(",") if t]
        sets = [_match_method(module, t, state_ref=state_ref) for t in tokens]
        group_set = sets[0]
        for s in sets[1:]:
            group_set &= s
        result |= group_set
    return result


def select(module: Module, expr: str, *, exclude: str = "", state_ref: str = "HEAD") -> list[Agent]:
    chosen = _eval_expr(module, expr, state_ref=state_ref)
    if exclude.strip():
        chosen -= _eval_expr(module, exclude, state_ref=state_ref)
    return sorted((a for a in module.agents if a.name in chosen), key=lambda a: a.name)
```

```python
@app.command("graph")
def graph(selector: str = typer.Argument("", help="Selection expression (default: all).")) -> None:
    """Render the cross-agent dependency DAG as Graphviz DOT."""
    module = _module()
    chosen = {a.name for a in select(module, selector)}
    lines = ["digraph agents {", '  rankdir=LR;']
    for a in module.agents:
        if a.name in chosen:
            lines.append(f'  "{a.name}";')
    for a in module.agents:
        if a.name not in chosen:
            continue
        for callee in a.calls:               # a calls callee -> edge callee -> a
            if callee in chosen:
                lines.append(f'  "{callee}" -> "{a.name}";')
    lines.append("}")
    typer.echo("\n".join(lines))
```

```python
# composable_agents/ca/tracetree.py
from __future__ import annotations

from composable_agents.projection import ProjectionEvent, SpanData, to_otel_spans


def render_tree(events: list[ProjectionEvent]) -> str:
    """Render projection events as an indented terminal trace tree (offline, no network)."""
    spans = to_otel_spans(events)
    by_cid: dict[str, SpanData] = {s.cid: s for s in spans}
    children: dict[str | None, list[SpanData]] = {}
    for s in spans:
        primary = s.parents[0] if s.parents else None
        parent = primary if primary in by_cid else None
        children.setdefault(parent, []).append(s)

    lines: list[str] = []

    def emit(cid_parent: str | None, prefix: str) -> None:
        kids = children.get(cid_parent, [])
        for i, span in enumerate(kids):
            last = i == len(kids) - 1
            branch = "└─ " if last else "├─ "
            cost = f" ${span.cost:.4f}" if span.cost else ""
            lines.append(f"{prefix}{branch}{span.node} [{span.status}]{cost}")
            emit(span.cid, prefix + ("   " if last else "│  "))

    emit(None, "")
    return "\n".join(lines)
```

```python
from dataclasses import dataclass, field
from typing import Any

from composable_agents.execution.interpreter import InMemoryEnv, interpret
from composable_agents.ir import Node
from composable_agents.projection import InMemoryProjection, ProjectionEmitter, ProjectionEvent

@dataclass(frozen=True)
class RunOutcome:
    run_id: str
    value: Any
    events: list[ProjectionEvent] = field(default_factory=list)
    error: str | None = None


def _echo_reasoner(_name: str, payload: Any) -> Any:
    """Default dev reasoner: returns a terminal output so the loop completes."""
    return {"output": payload}


def run_agent_local(resolved: ResolvedAgent, value: Any, *, run_id: str) -> RunOutcome:
    """Execute a resolved agent in-memory, owning the projection so we can return events."""
    if resolved.error is not None:
        return RunOutcome(run_id=run_id, value=None, error=resolved.error)
    node = Node.from_json(resolved.ir)
    projection = InMemoryProjection()
    emitter = ProjectionEmitter(projection)
    from composable_agents.contracts import ToolManifest  # local import avoids heavy top-level cost

    env = InMemoryEnv(ToolManifest(tools={}), emitter, reasoners={}, mode="dev")

    async def _go() -> Any:
        return await interpret(node, value, env)

    try:
        result = asyncio.run(_go())
    except Exception as exc:  # noqa: BLE001 — surface runtime errors as outcome, not crash
        return RunOutcome(run_id=run_id, value=None, events=projection.events(), error=str(exc))
    return RunOutcome(run_id=run_id, value=result.value, events=projection.events(), error=None)
```

```python
# composable_agents/ca/lint.py
from __future__ import annotations

from dataclasses import dataclass

from composable_agents.ca.config import JulepConfig
from composable_agents.ca.resolve import resolve_agent
from composable_agents.ir import Node
from composable_agents.validate import Diagnostic, validate

_SEVERITY_ORDER = {"info": 0, "warning": 1, "error": 2}


@dataclass(frozen=True)
class Finding:
    agent: str
    code: str
    severity: str
    message: str


def lint_agents(cfg: JulepConfig, names: list[str], *, fail_severity: str) -> tuple[list[Finding], int]:
    """Resolve each agent to IR, run validate(), collect findings, gate by fail_severity.

    Exit codes: 0 clean (or all below threshold) · 1 findings at/above threshold · 2 resolver error.
    """
    findings: list[Finding] = []
    floor = _SEVERITY_ORDER.get(fail_severity, 2)
    gated = False
    for name in names:
        resolved = resolve_agent(cfg, name)
        if resolved.error is not None:
            return [Finding(name, "RESOLVE", "error", resolved.error)], 2
        diags: list[Diagnostic] = validate(Node.from_json(resolved.ir))
        for d in diags:
            findings.append(Finding(name, d.code, d.severity, d.message))
            if _SEVERITY_ORDER.get(d.severity, 2) >= floor:
                gated = True
    return findings, (1 if gated else 0)
```

```python
from composable_agents.execution.langfuse import trace_id_for


def trace_url(run_id: str) -> str | None:
    """Best-effort clickable Langfuse trace URL from the stable trace id. None if unconfigured."""
    host = os.environ.get("LANGFUSE_HOST")
```

```python
# composable_agents/ca/doctor.py
from __future__ import annotations

import importlib.util
import os
import shutil
from dataclasses import dataclass

from composable_agents.ca.config import JulepConfig
from composable_agents.ca.model import build_module


@dataclass(frozen=True)
class Check:
    name: str
    ok: bool
    detail: str


def run_checks(cfg: JulepConfig) -> list[Check]:
    checks: list[Check] = []

    module = build_module(cfg)
    n = len(module.agents)
    checks.append(Check("discovery", n > 0, f"{n} agent(s) discovered under {cfg.src}"))

    has_git = shutil.which("git") is not None
    checks.append(Check("git", has_git, "git found" if has_git else "git not on PATH (state:modified disabled)"))

    lf = all(os.environ.get(k) for k in ("LANGFUSE_HOST",))
    checks.append(Check("langfuse", lf, "LANGFUSE_HOST set" if lf else "LANGFUSE_HOST unset (no deep links)"))

    has_temporal = importlib.util.find_spec("temporalio") is not None
    checks.append(Check("temporal", has_temporal, "temporalio installed" if has_temporal else "temporalio missing (deploy disabled)"))

    return checks


def overall_code(checks: list[Check]) -> int:
    """Exit 1 only if a hard-required check (discovery) fails; soft checks are warnings."""
    discovery = next((c for c in checks if c.name == "discovery"), None)
    return 0 if (discovery is None or discovery.ok) else 1
```
